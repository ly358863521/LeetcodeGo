class Solution(object):
    def spiralOrder(self, matrix):
        """
        :type matrix: List[List[int]]
        :rtype: List[int]
        """
        r, i, j, di, dj = [], 0, 0, 0, 1
        if matrix != []:
            for _ in range(len(matrix) * len(matrix[0])):
                r.append(matrix[i][j])
                matrix[i][j] = 100000
                if matrix[(i + di) % len(matrix)][(j + dj) % len(matrix[0])] == 100000:
                    di, dj = dj, -di
                i += di
                j += dj
        return r
